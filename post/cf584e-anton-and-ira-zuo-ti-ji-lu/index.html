#include <iostream>
#include <cstdio>

using namespace std;

const int S=2005;

struct node
{
	int x,y;
}res[S*S];

int n,a[S],b[S];
int pos[S],p[S],id[S];
int tot;

int main()
{
	scanf("%d",&n);
	for(int i=1;i<=n;i++) scanf("%d",&a[i]);
	for(int i=1;i<=n;i++) scanf("%d",&b[i]);
	for(int i=1;i<=n;i++) pos[b[i]]=i;
	int ans=0;
	for(int i=1;i<=n;i++) ans+=abs((p[i]=pos[a[i]])-i);
	for(int i=1;i<=n;i++) id[p[i]]=i;
	for(int i=n;i>=1;i--)
	{
		int pp=id[i];
		for(int j=pp+1;j<=i;j++)
		{
			if(p[j]<=pp)
			{
				res[++tot]=(node){pp,j};
				swap(p[pp],p[j]);
				swap(id[p[pp]],id[p[j]]);
				pp=j;
			}
		}
	}
	printf("%d\n",ans/2);
	printf("%d\n",tot);
	for(int i=1;i<=tot;i++)
	{
		printf("%d %d\n",res[i].x,res[i].y);
	}
	return 0;
}
